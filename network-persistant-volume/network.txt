docker run -d --name mysql-db \
  --network=my_network \
  -e MYSQL_ROOT_PASSWORD=rootpassword \
  -e MYSQL_DATABASE=attendance_db \
  -p 3306:3306 \
  mysql:latest

# With persistent storage (volume):
docker run -d --name mysql-db \
  --network=my_network \
  -v mysql_data:/var/lib/mysql \       [#  Mount volume for persistent storage]
  -e MYSQL_ROOT_PASSWORD=rootpassword \
  -e MYSQL_DATABASE=attendance_db \
  -p 3306:3306 \
  mysql:latest
